  <script>
    // あなたのGASのウェブアプリURL
    const GAS_URL = "https://script.google.com/macros/s/AKfycbwPrxIeBo1lKulI1MSd_ukPOWYnkAErv1fN5g_WAGS8mM1jGbnfFbmVXPL5sv_xKFqVXw/exec";

    async function loadData() {
      try {
        // GASへデータを取得しに行く
        const response = await fetch(GAS_URL);
        if (!response.ok) throw new Error('ネットワークエラーが発生しました');
        
        const data = await response.json();
        
        const statusArea = document.getElementById('status-area');
        const mainStatus = document.getElementById('main-status');
        const detailMsg = document.getElementById('detail-message');
        const updateTime = document.getElementById('update-time');
        
        // 1. 面会ステータスの反映
        const s = data.status;
        detailMsg.innerText = s.detail || "";
        updateTime.innerText = "最終更新: " + (s.updateTime || "---");
        
        // ステータスの判定と色の切り替え [cite: 111, 112]
        if (s.mainStatus === "OK" || s.mainStatus === "面会可能") {
            mainStatus.innerText = "面会可能";
            statusArea.className = "status-box ok";
        } else {
            mainStatus.innerText = "面会不可";
            statusArea.className = "status-box ng";
        }

        // 2. 掲示板データの反映 [cite: 113, 114]
        const list = document.getElementById('bbs-list');
        list.innerHTML = "";
        if (data.bbs && data.bbs.length > 0) {
          data.bbs.forEach(function(post) {
            const div = document.createElement('div');
            div.className = 'bbs-item';
            div.innerHTML = '<span class="bbs-time">' + post.time + '</span>' + post.content;
            list.appendChild(div);
          });
        }
      } catch (error) {
        console.error('Fetch error:', error);
        document.getElementById('main-status').innerText = "データ取得エラー";
      }
    }

    // 初回読み込みと30秒ごとの自動更新
    loadData();
    setInterval(loadData, 30000);
  </script>
